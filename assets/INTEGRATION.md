## 2. Wiring Voice Clips

### Basic `<audio>` element approach

Add one hidden audio element to the page. Swap `src` and call `.play()` from JS:

```html
<!-- Add once, anywhere in <body> -->
<audio id="metaldude-voice" preload="none"></audio>
```

```js
// Utility — call this whenever Metal Dude should speak
function metalDudeSay(clipName) {
  const el = document.getElementById('metaldude-voice');
  if (!el) return;
  el.src = `assets/voice/${clipName}.wav`;
  el.load();
  el.play().catch(() => {}); // Silently swallow autoplay blocks
}
```

### When to call it

```js
// On app ready / first user interaction (browsers require a gesture before audio):
metalDudeSay('metaldude_greeting');

// When user hits "Compose" / "Generate" — check current theme:
const theme = document.body.dataset.theme; // 'noir' or 'west'
metalDudeSay(theme === 'west' ? 'metaldude_compose_start_west' : 'metaldude_compose_start_noir');

// When mood/vibe changes:
const moodMap = {
  doom:    'metaldude_doom_mode',
  heavy:   'metaldude_doom_mode',
  dreamy:  'metaldude_dreamy_mode',
  ethereal:'metaldude_dreamy_mode',
  epic:    'metaldude_epic_mode',
  cinematic:'metaldude_epic_mode',
  // NOIR/WEST theme-specific handled above
};
if (moodMap[selectedMood]) metalDudeSay(moodMap[selectedMood]);

// When composition finishes rendering and the play button appears:
metalDudeSay('metaldude_compose_complete');

// When user clicks play on the generated riff:
metalDudeSay('metaldude_nice_riff');
```

### Volume / overlap control

```js
function metalDudeSay(clipName) {
  const el = document.getElementById('metaldude-voice');
  if (!el) return;
  el.pause();               // Stop any current clip
  el.currentTime = 0;
  el.volume = 0.75;         // Don't drown out the music
  el.src = `assets/voice/${clipName}.wav`;
  el.load();
  el.play().catch(() => {});
}
```

### How to wire covers into the UI (once generated)

Planned filenames and mood mapping:

```js
const COVER_MAP = {
  noir:       'assets/covers/cover_noir.png',
  west:       'assets/covers/cover_west.png',
  doom:       'assets/covers/cover_doom.png',
  heavy:      'assets/covers/cover_doom.png',
  dreamy:     'assets/covers/cover_dreamy.png',
  ethereal:   'assets/covers/cover_dreamy.png',
  epic:       'assets/covers/cover_epic.png',
  cinematic:  'assets/covers/cover_epic.png',
  melancholy: 'assets/covers/cover_melancholy.png',
  sad:        'assets/covers/cover_melancholy.png',
};
```

Display in the Composer result panel:

```html
<!-- Add inside your composition result / playback panel -->
<img id="track-cover" src="" alt="Track cover art"
     style="width:256px;height:256px;object-fit:cover;display:none;border-radius:8px;" />
```

```js
function showCover(mood) {
  const img = document.getElementById('track-cover');
  const src = COVER_MAP[mood] || COVER_MAP['noir']; // fallback to noir
  img.src = src;
  img.style.display = 'block';
}

// Call after composition completes:
showCover(currentMood);
```
